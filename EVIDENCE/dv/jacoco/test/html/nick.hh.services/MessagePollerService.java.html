package nick.hh.services;

import java.util.List;
import java.util.stream.Collectors;
import lombok.extern.slf4j.Slf4j;
import nick.hh.dtos.negotiations.NegotiationMessage;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;

/**
 * Получение резюме.
 */
@Slf4j
@Service
public class MessagePollerService {

    private final HhApiService hhApiService;

    @Value("${hh.messages.max.negotiations:5}")
    private int maxNegotiations;

    @Value("${hh.messages.log.limit:20}")
    private int logLimit;

    /**
     * Получение резюме.
     */
    public MessagePollerService(HhApiService hhApiService) {
        this.hhApiService = hhApiService;
    }

    /**
     * Получение резюме.
     */
    @Scheduled(fixedRateString = "${hh.messages.poll.interval:60000}")
    public void pollLatestMessages() {
        if (!hhApiService.hasToken()) {
            log.warn("Skipping message polling: no access token");
            return;
        }

        try {
            // Получаем все сообщения из первых 5 откликов с нужными статусами
            List<NegotiationMessage> messages = hhApiService.getLatestMessages(maxNegotiations);

            List<NegotiationMessage> messagesToLog = messages.stream()
                .limit(logLimit)
                .collect(Collectors.toList());

            logMessages(messagesToLog, messages.size());
        } catch (Exception e) {
            log.error("Message polling failed", e);
        }
    }

    private void logMessages(List<NegotiationMessage> messages, int totalCount) {
        if (messages.isEmpty()) {
            log.info("No new messages found");
            return;
        }

        log.info("--- Latest {} messages (of {} total) ---", messages.size(), totalCount);
        for (int i = 0; i < messages.size(); i++) {
            NegotiationMessage message = messages.get(i);

            String author = "employer".equals(message.getAuthor().getParticipantType())
                ? "Employer" : "You";

            String text = message.getText();
            if (text != null) {
                text = text.replaceAll("\n", " ").trim();
                if (text.length() > 80) {
                    text = text.substring(0, 80) + "...";
                }
            } else {
                text = "[NO TEXT]";
            }

            String time = message.getCreatedAt() != null
                ? message.getCreatedAt().toLocalTime().toString()
                : "[NO TIME]";

            log.info("{}. [{}][{}]: {}", i + 1, time, author, text);
        }

        if (totalCount > messages.size()) {
            log.info("... and {} more messages not shown", totalCount - messages.size());
        }

        log.info("--- End of messages ---");
    }
}
