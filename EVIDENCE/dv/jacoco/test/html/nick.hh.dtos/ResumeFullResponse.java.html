package nick.hh.dtos;

import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
import com.fasterxml.jackson.annotation.JsonProperty;
import java.util.List;
import java.util.Map;
import lombok.Data;

/**
 * Ответ на резюме.
 */
@JsonIgnoreProperties(ignoreUnknown = true)
@Data
public class ResumeFullResponse {
    private String id;
    private String title;
    private String skills;

    @JsonProperty("skill_set")
    private List<String> skillSet;

    @JsonProperty("first_name")
    private String firstName;

    @JsonProperty("last_name")
    private String lastName;

    @JsonProperty("middle_name")
    private String middleName;

    private String description;

    @JsonProperty("total_experience")
    private TotalExperience totalExperience;

    private List<Experience> experience;
    private Education education;
    private List<Contact> contact;

    @JsonProperty("professional_roles")
    private List<ProfessionalRole> professionalRoles;


    /**
     * Опыт.
     */
    @Data
    public static class TotalExperience {
        private Integer months;
    }

    /**
     * Опыт.
     */
    @Data
    public static class Experience {
        private String company;
        private String position;
        private String description;
        private String start;
        private String end;
        private Area area;

        /**
         * Область.
         */
        @Data
        public static class Area {
            private String id;
            private String name;
        }
    }

    /**
     * Образование.
     */
    @Data
    public static class Education {
        private Level level;

        @JsonProperty("primary")
        private List<EducationItem> primaryEducation;

        @JsonProperty("additional")
        private List<EducationItem> additionalEducation;

        /**
         * Уровень.
         */
        @Data
        public static class Level {
            private String id;
            private String name;
        }

        /**
         * Образование.
         */
        @Data
        public static class EducationItem {
            private String name;
            private String organization;
            private String result;
            private Integer year;
        }
    }

    /**
     * Контакт.
     */
    @Data
    public static class Contact {
        private Type type;
        private Object value; // Может быть String или Map
        private Boolean preferred;

        /**
         * Тип.
         */
        @Data
        public static class Type {
            private String id;
            private String name;
        }

        /**
         * Получение стринги.
         */
        public String getValueAsString() {
            if (value instanceof String) {
                return (String) value;
            } else if (value instanceof Map) {
                Map<?, ?> map = (Map<?, ?>) value;
                if (map.containsKey("formatted")) {
                    return map.get("formatted").toString();
                } else if (map.containsKey("value")) {
                    return map.get("value").toString();
                }
                return value.toString();
            }
            return value != null ? value.toString() : null;
        }
    }

    /**
     * Роль.
     */
    @Data
    public static class ProfessionalRole {
        private String id;
        private String name;
    }
}
